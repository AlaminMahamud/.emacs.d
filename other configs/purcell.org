* Dynamic Binding Vs Lexical Binding
#+BEGIN_SRC emacs-lisp
  (let ((a 1))
    (print a))
#+END_SRC

#+BEGIN_SRC emacs-lisp
  (let ((a 1))
    (let ((a 2))
      (let ((a 3))
        (print a))
      (print a))
    (print a))
#+END_SRC

#+begin_src emacs-lisp
  (defun foo (a)
    (let ((a 2))
      (print a))
    (print a))
  (foo 1)
#+end_src
